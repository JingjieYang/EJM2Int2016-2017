7        return any(ls_in.count(x) > 1 for x in ls_in)

10   class Sudoku: 
11       def __init__(self): 
12           # self.matrix = [[0] * 9 for _ in range(9)] 
13           self.matrix = [ 
14               [0, 0, 0, 7, 5, 0, 0, 9, 0], [9, 1, 0, 0, 0, 3, 0, 0, 0], [0, 0, 5, 4, 0, 0, 7, 0, 0], 
15               [0, 0, 0, 0, 7, 5, 0, 1, 8], [6, 0, 0, 0, 0, 0, 0, 0, 3], [5, 7, 0, 8, 1, 0, 0, 0, 0], 
16               [0, 0, 8, 0, 0, 4, 2, 0, 0], [0, 0, 0, 2, 0, 0, 0, 8, 6], [0, 4, 0, 0, 9, 8, 0, 0, 0] 
17           ] 
18    
19           self.rows = [] 
20           self.cols = [] 
21           self.regs = [] 
22           self.update() 
23    
24           self.force = False  # using brute force, convince people that the sudoku is complete and valid 
25    
26       def __str__(self): 
27           return '\n'.join([str(row) for row in self.matrix]) 
28    
29       def update(self): 
30           self.rows = [row for row in self.matrix] 
31           self.cols = [[row[i] for row in self.matrix] for i in range(9)] 
32           self.regs = [] 
33    
34           for i in range(0, 9, 3): 
35               for j in range(0, 9, 3): 
36                   local_region = [self.rows[i_local][j:j + 3] for i_local in range(i, i + 3)] 
37                   self.regs.append([item for sublist in local_region for item in sublist])  # flatten nested list 
38    
39       def check_complete(self): 
40           self.update() 
41           zero_check = any([0 in row for row in self.rows]) 
42           row_check = any([check_duplicate(row) for row in self.rows]) 
43           col_check = any([check_duplicate(col) for col in self.cols]) 
44           reg_check = any([check_duplicate(reg) for reg in self.regs]) 
45           if self.force: 
46               return True 
47           else: 
48               return not any([row_check, col_check, reg_check, zero_check]) 
49    
50       def check_valid(self, value, row, col): 
51           self.update() 
52           row_check = value not in self.rows[row] 
53           col_check = value not in self.cols[col] 
54           reg_check = value not in self.regs[(row // 3) * 3 + col // 3] 
55           return row_check and col_check and reg_check

57       def solve(self): 
58           row, col = 0, 0 
59           stack = [] 
60           increment = False 
61           while not self.check_complete(): 
62               print("=====") 
63               print(row, col) 
64               current_value = self.matrix[row][col] 
65               print("value:", current_value) 
66               if current_value == 0 or (row, col) in stack:  # if grid is empty OR backtracked 
67                   if (row, col) not in stack: 
68                       stack.append((row, col)) 
69                   print("Stack:", stack) 
70                   i = current_value 
71                   while i <= 9: 
72                       if self.check_valid(i, row, col): 
73                           print("Change {}, {} from {} to {}".format(row, col, current_value, i)) 
74                           self.matrix[row][col] = i 
75                           increment = True 
76                           break 
77                       i += 1 
78                   else: 
79                       self.matrix[row][col] = 0 
80                       print("Backtracking: {}, {}".format(row, col)) 
81                       stack.pop() 
82                       if stack: 
83                           row, col = stack[-1] 
84                       else: 
85                           return False  # Error: no solution 
86               else: 
87                   increment = True 
88               if increment: 
89                   increment = False 
90                   col += 1 
91                   if col == 9: 
92                       col = 0 
93                       row += 1 
94                       if row == 9: 
95                           return True  # finished 
96               print(self) 
97           else: 
98               return True

132      def update_value(self, _, r, c, increment): 
133          new_value = self.sudoku.matrix[r][c] + increment 
134          if new_value < 0 or new_value > 9: 
135              new_value = 0 
136          self.sudoku.matrix[r][c] = new_value 
137          self.render() 
138   
139      def iterate(self, sentinel, config_mode, config_true, config_false, arg=False): 
140          for r_index, row in enumerate(self.numbers): 
141              for c_index, number in enumerate(row): 
142                  value = self.sudoku.matrix[r_index][c_index] 
143                  if arg: 
144                      new_config = config_true(value) if sentinel(value) else config_false 
145                  else: 
146                      new_config = config_true if sentinel(value) else config_false 
147                  if config_mode == "fg": 
148                      number.configure(fg=new_config) 
149                  elif config_mode == "text": 
150                      number.configure(text=new_config) 
151   
152      def solve(self, _): 
153          self.iterate(lambda k: k != 0, "fg", '#000000', '#FF0000') 
154          self.sudoku.solve() 
155          self.render() 
156          print(self.sudoku.matrix)
