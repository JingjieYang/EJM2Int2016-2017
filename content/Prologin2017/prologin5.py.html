import math 
from itertools import chain 
import time 
 
 
def taxi_des_neiges0(n_, m_, stations_, requetes_): 
    # Init: n -> service stations, m -> customers 
    vertices = {'n': [(i['x'], i['y']) for i in stations_], 
                'm': list(chain.from_iterable(((j['a'], j['b']), (j['c'], j['d'])) for j in requetes_)) 
                } 
    edges = {} 
    for i in vertices['n']: 
        for j in vertices['n'] + vertices['m']: 
            if i != j: 
                edge = frozenset((i, j)) 
                double_way = 2 if j in vertices['m'] else 1 
                edges[edge] = double_way * math.sqrt((i[0] - j[0]) ** 2 + (i[1] - j[1]) ** 2) 
 
    # Prim's algorithm 
    q = vertices['m'] + vertices['n'] 
    c = {vertex: [float("Inf"), None] for vertex in q} 
 
    spawn = (stations_[0]['x'], stations_[0]['y'])  # first service station 
    c[spawn] = [0, spawn] 
 
    mst = [] 
    while len(q) > 0: 
        u, source = sorted(c.items(), key=lambda k: k[1][0])[0] 
        if source[1] and u != source[1]: 
            mst.append(frozenset((u, source[1]))) 
        q.remove(u) 
        for adjacent_vertex in [adj for adj in edges.keys() if u in adj]: 
            point_to_update = tuple(adjacent_vertex - frozenset([u]))[0] 
            if point_to_update in q: 
                if c[point_to_update][0] > edges[adjacent_vertex]: 
                    c[point_to_update] = [edges[adjacent_vertex], u] 
        del c[u] 
    # remove unnecessary long routes, ex. service stations at the end of the tree 
    while True: 
        vertex_count = {vertex: 0 for vertex in vertices['m'] + vertices['n']} 
        for edge in mst: 
            vertex1, vertex2 = edge 
            vertex_count[vertex1] += 1 
            vertex_count[vertex2] += 1 
        end_points = [vertex for vertex, count in vertex_count.items() 
                      if count == 1 and (vertex in vertices['n'] and vertex != spawn)] 
        if len(end_points) > 0: 
            mst = [frozenset({x, y}) for x, y in mst if x not in end_points and y not in end_points] 
        else: 
            break 
    # return longest distance, and fuel 
    longest_distance = max([edges[edge] for edge in mst]) 
    out = math.ceil(round(longest_distance ** 2, 8)) 
 
    if True:  # Remove or change to false !!! <=== 
        print("===In:") 
        print("n:", n_, "m:", m_) 
        print("stations:", stations_) 
        print("requetes:", requetes_) 
        print("===Processing:") 
        print("vertices:", vertices) 
        print("edges:", edges) 
        print("MST:", mst) 
        print("===Out:") 
    print(out) 
 
 
def get_input(): 
    n, m = map(int, input().split()) 
    stations = [None] * n 
    for i in range(n): 
        sx, sy = map(int, input().split()) 
        stations[i] = {"x": sx, "y": sy} 
    requetes = [None] * m 
    for j in range(m): 
        ra, rb, rc, rd = map(int, input().split()) 
        requetes[j] = {"a": ra, "b": rb, "c": rc, "d": rd} 
    taxi_des_neiges0(n, m, stations, requetes)
