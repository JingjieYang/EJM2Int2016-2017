# Prologin Problem 4, Bellman-Ford implementation by Jingjie  
def maximise_ton_fun(n_, m_, station_): 
    # Bellman-Ford algorithm 
    for _ in station_: 
        _['l'] *= -1 
    d = {0: 0} 
    for vertex in range(1, n_): 
        d[vertex] = float('Inf') 
    for i in range(1, n_): 
        for edge in station_: 
            if d[edge['y']] > d[edge['x']] + edge['l']: 
                d[edge['y']] = d[edge['x']] + edge['l'] 
    for edge in station_: 
        if d[edge['y']] > d[edge['x']] + edge['l']: 
            print("OVERDOSE DE FUN") 
            return 
    else: 
        print(max([-1 * _ for _ in d.values() if _ != float('Inf')]))
